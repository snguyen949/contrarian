Entry / Exit Rules — Pure Supertrend

BUY
Supertrend flips green — candle closes above the ST line after being below it. Enter at that close price.
SELL
Supertrend flips red — candle closes below the ST line after being above it. Enter short at that close.
STOP
Stop = Supertrend line value at entry bar. Dynamic — tightens automatically as the ST line trails price during the trade.
EXIT
Partial exits: 25% at T1 (2R), 50% at T2 (3.5R). Remaining 25% trails until the opposite ST flip closes the position.
GATE
Skip if R:R < 2.0 at T1. No confluence filter — purely the Supertrend signal on one chart.

Pine Script v5 — Supertrend Strategy (Single Ticker · 4H)

// ─────────────────────────────────────────────────────────────────────────────
// SUPERTREND STRATEGY  |  Pine Script v5  |  Single Ticker
// Apply to any chart: MES, SPY, QQQ, NQ, BTC/USD, etc.
// BUY  → Supertrend flips bullish (price closes above ST line)
// SELL → Supertrend flips bearish (price closes below ST line)
// STOP → Supertrend line value at entry (dynamic trailing stop)
// ─────────────────────────────────────────────────────────────────────────────
//@version=5
strategy("Supertrend Strategy", overlay=true,
         default_qty_type=strategy.fixed, default_qty_value=1,
         initial_capital=50000, commission_value=2.0)

// ── INPUTS ────────────────────────────────────────────────────────────────────
atr_len  = input.int(10,   title="ATR Length",       minval=1)
factor   = input.float(3.0, title="Factor",            minval=0.5, step=0.1)
t1_r     = input.float(2.0, title="T1 R Multiple",     step=0.5)
t2_r     = input.float(3.5, title="T2 R Multiple",     step=0.5)
rth_only = input.bool(true, title="RTH Session Only")
i_start  = input.time(timestamp("2024-01-01"), "Start Date")

// ── SUPERTREND ────────────────────────────────────────────────────────────────
atr  = ta.atr(atr_len)
up   = hl2 - factor * atr
dn   = hl2 + factor * atr
up  := close[1] > nz(up[1]) ? math.max(up, nz(up[1])) : up
dn  := close[1] < nz(dn[1]) ? math.min(dn, nz(dn[1])) : dn
trend  = 1
trend := close > nz(dn[1]) ? 1 : close < nz(up[1]) ? -1 : nz(trend[1], 1)
st     = trend == 1 ? up : dn

bull_flip = trend ==  1 and trend[1] == -1  // green flip  → BUY
bear_flip = trend == -1 and trend[1] ==  1  // red flip    → SELL SHORT

// ── FILTERS ──────────────────────────────────────────────────────────────────
in_session = not rth_only or
              time(timeframe.period, "0930-1600", "America/New_York") != 0
in_range   = time >= i_start
ok         = in_session and in_range

// ── STATE ─────────────────────────────────────────────────────────────────────
var float ep   = na    // entry price
var float er   = na    // entry risk (entry - stop)
var bool  t1ok = false
var bool  t2ok = false

// ── ENTRIES ───────────────────────────────────────────────────────────────────
if bull_flip and ok and strategy.position_size == 0
    ep   := close
    er   := math.abs(close - st)
    t1ok := false
    t2ok := false
    strategy.entry("Long", strategy.long)
    alert("🟢 BUY · Supertrend flipped BULLISH · " + str.tostring(close), alert.freq_once_per_bar)

if bear_flip and ok and strategy.position_size == 0
    ep   := close
    er   := math.abs(close - st)
    t1ok := false
    t2ok := false
    strategy.entry("Short", strategy.short)
    alert("🔴 SELL · Supertrend flipped BEARISH · " + str.tostring(close), alert.freq_once_per_bar)

// ── PARTIAL EXITS (Long) ──────────────────────────────────────────────────────
if strategy.position_size > 0 and not na(ep)
    if not t1ok and high >= ep + t1_r * er
        strategy.exit("L-T1", "Long", qty_percent=25, limit=ep + t1_r * er)
        t1ok := true
    if t1ok and not t2ok and high >= ep + t2_r * er
        strategy.exit("L-T2", "Long", qty_percent=67, limit=ep + t2_r * er)
        t2ok := true

// ── PARTIAL EXITS (Short) ─────────────────────────────────────────────────────
if strategy.position_size < 0 and not na(ep)
    if not t1ok and low <= ep - t1_r * er
        strategy.exit("S-T1", "Short", qty_percent=25, limit=ep - t1_r * er)
        t1ok := true
    if t1ok and not t2ok and low <= ep - t2_r * er
        strategy.exit("S-T2", "Short", qty_percent=67, limit=ep - t2_r * er)
        t2ok := true

// ── FULL EXIT ON OPPOSITE FLIP ────────────────────────────────────────────────
if strategy.position_size > 0 and bear_flip
    strategy.close("Long")
if strategy.position_size < 0 and bull_flip
    strategy.close("Short")

// ── PLOTS ─────────────────────────────────────────────────────────────────────
plot(st, title="Supertrend", linewidth=2,
     color=trend==1 ? color.new(color.teal,0) : color.new(color.red,0))
bgcolor(trend==1 ? color.new(color.teal,97) : color.new(color.red,97))
plotshape(bull_flip, style=shape.triangleup,   location=location.belowbar,
          color=color.teal, size=size.small, title="Buy")
plotshape(bear_flip, style=shape.triangledown, location=location.abovebar,
          color=color.red,  size=size.small, title="Sell")
